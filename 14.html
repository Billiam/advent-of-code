  if mask
    chars = mask.captures.first.split("")
    on_mask = chars.map {|i| i=="1" ? 1:0}.join("").to_i(2)

    key = key | on_mask

      new_key = key
      float_mask.zip(float_set).each do |position, value|

          new_key &= (~(1 << position))
        end
      end
      mem2[new_key]=val

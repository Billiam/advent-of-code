  s = StringScanner.new(v)
  stack = [0]
  mod = nil
  
  until s.eos?
    case
      when open = s.scan('(')
        stack << (mod || '+')
        stack << 0
        mod = nil
      when s.scan(')')
        last_mod, paren_result = stack.pop(2)
        stack[-1] = stack.last.send(last_mod.to_sym, paren_result)
      when m = s.scan(/[+-\/*]/)
        mod = m
      when n = s.scan(/\d+/)
        if mod
          stack[-1] = stack.last.send(mod.to_sym, n.to_i)
        else
         stack[-1] += n.to_i
       end
       mod = nil
    else
    end
  end
  stack.first
end.inject(:+)

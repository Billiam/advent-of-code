def run(data)
  acc=0
  ins=0
  lines = []

  while !lines.include?(ins)
    return [nil, acc] unless data[ins]

    lines << ins
    cmd, num = data[ins]
    
    ins += cmd == "jmp" ? num : 1
    acc += num if cmd == "acc"
  end

  [lines, acc]
end

  d = data.dup
  cmd,num = data[line_num]
  next if cmd == "acc"
  
  d[line_num] = [cmd == "nop" ? "jmp" : "nop", num]

  l, acc = run(d) 
  break acc if l.nil?
end
